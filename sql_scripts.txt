CREATE TABLE ProjectContract (
    Id UNIQUEIDENTIFIER PRIMARY KEY,
    ProjectId INT,
    ContractModelType INT,
    ContractTypeID INT,
    ContractTypeName NVARCHAR(MAX),
    CurrencyID INT,
    CurrencyISO NVARCHAR(3),
    CustomerID INT,
    ExpenseBudgetAmount DECIMAL(18, 2),
    ExpenseRevenueShareAmount DECIMAL(18, 2),
    ExternalWorkRevenueShareAmount DECIMAL(18, 2),
    GoodsRevenueShareAmount DECIMAL(18, 2),
    LastModified DATETIME,
    Name NVARCHAR(MAX),
    PaymentAmount DECIMAL(18, 2),
    ProjectContractStatus INT,
    RiskRevenueShareAmount DECIMAL(18, 2),
    TravelBudgetAmount DECIMAL(18, 2),
    TravelRevenueShareAmount DECIMAL(18, 2),
    WorkBudgetHours DECIMAL(18, 2),
    WorkRevenueShareAmount DECIMAL(18, 2),
    CreatedOn DATETIME NOT NULL,
    ModifiedOn DATETIME,
    CreatedBy NVARCHAR(MAX),
    ModifiedBy NVARCHAR(MAX)
);

CREATE TABLE ProjectExpenses (
    Id UNIQUEIDENTIFIER PRIMARY KEY,
    ProjectExpenseID INT NOT NULL DEFAULT(0),
    ExpenseNo NVARCHAR(50),
    ProjectID INT NOT NULL DEFAULT(0),
    EmployeeID INT DEFAULT(0),
    PurchaseDate DATETIME,
    InvoiceNo NVARCHAR(50),
    CustomerID INT DEFAULT(0),
    ExpenseCurrencyISO NVARCHAR(3),
    ExpenseDirectProjectExchangeRate FLOAT DEFAULT(0),
    AmountExpenseCurrency FLOAT DEFAULT(0),
    VatExpenseCurrency FLOAT DEFAULT(0),
    ProjectCurrencyISO NVARCHAR(3),
    SalesPriceProjectCurrency FLOAT DEFAULT(0),
    SalesPriceSystemCurrency FLOAT DEFAULT(0),
    SalesPriceProjectExchangeRate FLOAT DEFAULT(0),
    ExpenseCostProjectCurrency FLOAT DEFAULT(0),
    ExpenseCostSystemCurrency FLOAT DEFAULT(0),
    ProfitRatio FLOAT DEFAULT(0),
    InvoicedAmountSystemCurrency FLOAT DEFAULT(0),
    InvoicedAmountProjectCurrency FLOAT DEFAULT(0),
    Comment NVARCHAR(MAX),
    PaymentMethodID INT DEFAULT(0),
    ExpenseTypeID INT DEFAULT(0),
    IsBillable BIT DEFAULT(0),
    IsApproved BIT DEFAULT(0),
    ApprovedBy INT DEFAULT(0),
    InvoiceLineID INT DEFAULT(0),
    UnitType INT DEFAULT(0),
    ProductNo NVARCHAR(50),
    DisbursementGuid NVARCHAR(50),
    ExternalID NVARCHAR(50),
    ProjectSubContractID INT DEFAULT(0),
    InstallmentID INT DEFAULT(0),
    LegalEntityID INT DEFAULT(0),
    CreatedOn DATETIME NOT NULL,
    ModifiedOn DATETIME NULL,
    CreatedBy NVARCHAR(100) NULL,
    ModifiedBy NVARCHAR(100) NULL
);
